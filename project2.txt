  <script>
    let editIndex = null;

    function calculateGrade(percentage) {
      if (percentage >= 75) return 'A';
      if (percentage >= 60) return 'B';
      if (percentage >= 50) return 'C';
      if (percentage >= 40) return 'D';
      return 'F';
    }

    function calculateResult(percentage) {
      return percentage >= 40 ? 'PASS' : 'FAIL';
    }

    function saveStudent() {
      const name = document.getElementById("name").value.trim();
      const roll = document.getElementById("roll").value.trim();
      const maths = parseFloat(document.getElementById("maths").value);
      const science = parseFloat(document.getElementById("science").value);
      const english = parseFloat(document.getElementById("english").value);
      const presentDays = parseFloat(document.getElementById("presentDays").value);
      const totalDays = parseFloat(document.getElementById("totalDays").value);

      if (!name || !roll || isNaN(maths) || isNaN(science) || isNaN(english) || isNaN(presentDays) || isNaN(totalDays) || totalDays <= 0) {
        alert("⚠️ Please fill all fields correctly!");
        return;
      }

      const total = maths + science + english;
      const percentage = total / 3;
      const attendance = (presentDays / totalDays) * 100;
      const grade = calculateGrade(percentage);
      const result = calculateResult(percentage);

      const student = { name, roll, total, percentage, grade, result, attendance };

      let students = JSON.parse(localStorage.getItem("students")) || [];

      if (editIndex !== null) {
        students[editIndex] = student;
        editIndex = null;
      } else {
        students.push(student);
      }

      localStorage.setItem("students", JSON.stringify(students));
      clearForm();
      loadStudents();
    }

    function loadStudents() {
      const tbody = document.querySelector("#studentTable tbody");
      tbody.innerHTML = "";
      const students = JSON.parse(localStorage.getItem("students")) || [];

      students.forEach((s, index) => {
        const tr = document.createElement("tr");
        tr.innerHTML = `
          <td>${s.name}</td>
          <td>${s.roll}</td>
          <td>${s.total}</td>
          <td>${s.percentage.toFixed(2)}%</td>
          <td>${s.grade}</td>
          <td class="${s.result === 'PASS' ? 'pass' : 'fail'}">${s.result}</td>
          <td>${s.attendance.toFixed(2)}%</td>
          <td>
            <button class="edit-btn" onclick="editStudent(${index})"><i class="fa-solid fa-pen"></i></button>
            <button class="delete-btn" onclick="deleteStudent(${index})"><i class="fa-solid fa-trash"></i></button>
          </td>
        `;
        tbody.appendChild(tr);
      });
    }

    function editStudent(index) {
      const students = JSON.parse(localStorage.getItem("students")) || [];
      const s = students[index];
      document.getElementById("name").value = s.name;
      document.getElementById("roll").value = s.roll;
      document.getElementById("maths").value = s.total / 3; // approximate split
      document.getElementById("science").value = s.total / 3;
      document.getElementById("english").value = s.total / 3;
      editIndex = index;
    }

    function deleteStudent(index) {
      if (confirm("Are you sure you want to delete this record?")) {
        let students = JSON.parse(localStorage.getItem("students")) || [];
        students.splice(index, 1);
        localStorage.setItem("students", JSON.stringify(students));
        loadStudents();
      }
    }

    function clearForm() {
      document.querySelectorAll("input").forEach(i => i.value = "");
    }

    function clearAll() {
      if (confirm("Delete all student records?")) {
        localStorage.removeItem("students");
        loadStudents();
      }
    }

    window.onload = loadStudents;
  </script>
